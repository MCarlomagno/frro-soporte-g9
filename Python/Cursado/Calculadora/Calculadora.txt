import tkinter as tk
from tkinter import *

def sumar():
    text3.set(text1.get() + text2.get())

def restar():
    text3.set(text1.get() - text2.get())

def dividir():
    text3.set(text1.get() / text2.get())

def multiplicar():
    text3.set(text1.get() * text2.get())

calculadora = Tk()
text1 = DoubleVar()
text2 = DoubleVar()
text3 = DoubleVar()

btnsumar = Button(text='Sumar',command=sumar)
btnrestar = Button(text='Restar',command=restar)
btndividir = Button(text='    Dividir    ',command=dividir)
btnmultiplicar= Button(text='Multiplicar',command=multiplicar)


oEntrada = Entry(calculadora, textvariable = text1 )
oEntrada2 = Entry(calculadora, textvariable = text2 )
label = Label(textvariable= text3)
text1.set("")
text2.set("")
text3.set("")

oEntrada.grid(column=0,row=0,sticky=(N, S, E, W))
oEntrada2.grid(column=0,row=1,sticky=(N, S, E, W))
btnsumar.grid(column=1,row=0,sticky=(N, S, E, W))
label.grid(column=0,row=2,sticky=(N, S, E, W))
btnrestar.grid(column=1,row=1,sticky=(N, S, E, W))
btndividir.grid(column=2,row=0,sticky=(N, S, E, W))
btnmultiplicar.grid(column=2,row=1,sticky=(N, S, E, W))
calculadora.columnconfigure(0,weight=1)
calculadora.columnconfigure(1,weight=1)
calculadora.columnconfigure(2,weight=1)
calculadora.rowconfigure(0,weight=1)
calculadora.rowconfigure(1,weight=1)
calculadora.rowconfigure(2,weight=1)



calculadora.mainloop()
